    import torch
import torch.nn as nn
import torch.optim as optim
import torchvision
import torchvision.transforms as transforms

# 定义超参数
batch_size = 64
learning_rate = 0.001
num_epochs = 5

# 数据预处理
transform = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))
])

# 加载MNIST数据集
train_dataset = torchvision.datasets.MNIST(root='./data', train=True, transform=transform, download=True)
test_dataset = torchvision.datasets.MNIST(root='./data', train=False, transform=transform, download=True)

train_loader = torch.utils.data.DataLoader(dataset=train_dataset, batch_size=batch_size, shuffle=True)
test_loader = torch.utils.data.DataLoader(dataset=test_dataset, batch_size=batch_size, shuffle=False)

# 定义神经网络模型
class Net(nn.Module):
    def __init__(self):
        super(Net, self).__init__()
        self.fc1 = nn.Linear(28*28, 500)# 第一个全连接层

        self.fc2 = nn.Linear(500, 10)# 第二个全连接层


    def forward(self, x):# 前向传播
        x = x.view(-1, 28*28)
  # MNIST数据集中的每个图像是28×28像素，表示为一个矩阵。全连接层的输入是一个向量，因此需要将矩阵“拉直”成一维向量.x.view(-1, 28*28): 将输入图像从28×28（二维）展平成一个大小为784（28*28）的向量。-1 表示该维度由PyTorch自动推断，因此如果输入是一个批量数据（如64张图片），view 会生成一个大小为 (64, 784) 的张量。

        x = torch.relu(self.fc1(x))
  # 输入 𝑥 经 fc1 变换后，将生成一个500维的向量，并通过ReLU激活，保留正值并将负值置为零，增加了网络的非线性拟合能力。    

        x = self.fc2(x)
        return x

model = Net()

# 定义损失函数和优化器

#在 PyTorch 中，损失函数通过 torch.nn 中的内置函数调用。例如在多分类任务中使用 CrossEntropyLoss
criterion = nn.CrossEntropyLoss()
optimizer = optim.Adam(model.parameters(), lr=learning_rate)

# 训练模型
for epoch in range(num_epochs):
    for i, (images, labels) in enumerate(train_loader):
        outputs = model(images)

        #反向传播
        loss = criterion(outputs, labels)    # 计算损失
        optimizer.zero_grad()               # 清除先前的梯度
        loss.backward()                     # 反向传播计算梯度
        optimizer.step()                    # 优化器更新参数

        if (i+1) % 100 == 0:
            print(f'Epoch [{epoch+1}/{num_epochs}], Step [{i+1}/{len(train_loader)}], Loss: {loss.item():.4f}')

# 测试模型
model.eval()
with torch.no_grad():
    correct = 0
    total = 0
    for images, labels in test_loader:
        outputs = model(images)
        _, predicted = torch.max(outputs.data, 1)
        total += labels.size(0)
        correct += (predicted == labels).sum().item()

    print(f'Accuracy of the model on the 10000 test images: {100 * correct / total} %')

    第一个全连接层

    输入和输出：28*28表示输入的784维向量（即展平的图像数据），500表示该层有500个输出神经元。

    参数个数：权重矩阵 \(\(W1\)\) 大小为 \(\(784×500\)\)，因此有 \((784×500=392,000 (\(个权重参数。同时有\)\)500\)\)个偏置参数（每个输出神经元一个偏置值），所以总参数量为 \(\(392,500\)\)。

    激活函数：网络通过ReLU激活函数引入非线性，避免多个全连接层的线性组合仅能表现出线性关系。

    第二个全连接层

    输入和输出：该层的输入大小为\(\(500\)\)（来自第一层的输出），输出大小为\(\(10\)\)，对应MNIST的\(\(10\)\)个分类（数字0到9）。

    参数个数：权重矩阵 \((W_2 (\(大小为\)\) 500 \times 10=5000\)\)，总共5,000个权重；偏置为10个。因此该层有 \(\(5,010\)\) 个参数。

    激活函数：第二层没有额外的激活函数。一般来说，最后一层的输出会直接传给损失函数 CrossEntropyLoss，该函数自带 Softmax操作，将每个输出数值转化为概率分布（代表每个类别的概率）。

    import os
import glob
import torch
from torch import nn, optim
from torch.utils.data import Dataset, DataLoader
from torchvision import transforms
from PIL import Image

# 自定义数据集类
class ArmorDataset(Dataset):
    def __init__(self, root_dir, transform=None):
        self.root_dir = root_dir
        self.transform = transform
        self.classes = sorted(os.listdir(root_dir))
        self.image_paths = []
        self.labels = []

        for label, class_name in enumerate(self.classes):
            class_dir = os.path.join(root_dir, class_name)
            for image_path in glob.glob(os.path.join(class_dir, '*')):
                self.image_paths.append(image_path)
                self.labels.append(label)

    def __len__(self):
        return len(self.image_paths)

    def __getitem__(self, idx):
        img_path = self.image_paths[idx]
        image = Image.open(img_path).convert('RGB')
        label = self.labels[idx]

        if self.transform:
            image = self.transform(image)

        return image, label

# 图像预处理
transform = transforms.Compose([
    transforms.Resize((64, 64)),
    transforms.ToTensor(),
    transforms.Normalize((0.5,), (0.5,))
])

# 创建数据集和数据加载器
dataset = ArmorDataset(root_dir='/home/chenyu/TIT/TIT_armor', transform=transform)
train_loader = DataLoader(dataset, batch_size=32, shuffle=True)

# 定义卷积神经网络模型
class ArmorClassifier(nn.Module):
    def __init__(self):
        super(ArmorClassifier, self).__init__()
        self.conv_layers = nn.Sequential(
            nn.Conv2d(3, 16, kernel_size=3, stride=1, padding=1),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2),

            nn.Conv2d(16, 32, kernel_size=3, stride=1, padding=1),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2),

            nn.Conv2d(32, 64, kernel_size=3, stride=1, padding=1),
            nn.ReLU(),
            nn.MaxPool2d(kernel_size=2, stride=2)
        )
        self.fc_layers = nn.Sequential(
            nn.Linear(64 * 8 * 8, 128),
            nn.ReLU(),
            nn.Linear(128, len(os.listdir('/home/chenyu/TIT/TIT_armor')))
        )

    def forward(self, x):
        x = self.conv_layers(x)
        x = x.view(x.size(0), -1)
        x = self.fc_layers(x)
        return x

# 初始化模型、损失函数和优化器
model = ArmorClassifier()
criterion = nn.CrossEntropyLoss()
optimizer = optim.Adam(model.parameters(), lr=0.001)

# 训练模块
def train_model(model, dataloader, criterion, optimizer, epochs=5):
    for epoch in range(epochs):
        model.train()
        running_loss = 0.0

        for images, labels in dataloader:
            optimizer.zero_grad()
            outputs = model(images)
            loss = criterion(outputs, labels)
            loss.backward()
            optimizer.step()
            running_loss += loss.item()

        print(f"Epoch {epoch + 1}/{epochs}, Loss: {running_loss / len(dataloader)}")

train_model(model, train_loader, criterion, optimizer)

    pytorch风格的数据集

    1. torch.utils.data.Dataset是 PyTorch提供的一个基础类，用于表示数据集。它是构建自定义数据集的基类，使用户能够以标准的接口和方式加载数据。这一类主要是为 PyTorch 的 DataLoader 设计的，通过实现两个必要的方法来定义数据集：__len__() 和 __getitem__()

    2. __len__()：返回数据集中样本的数量。

    __getitem__(index)：根据给定索引 index 获取一个样本的数据和标签，通常返回一个 (data, label) 元组

    ​ 通过这两个方法，Dataset 能够兼容 DataLoader，从而支持批量加载、并行处理、随机打乱数据等操作。

    卷积层

    卷积层使用小的卷积核（如 3x3），通过滑动窗口在图像上提取局部特征。每一层卷积后使用激活函数（ReLU）将非线性特征引入模型中，然后通过池化层（Max Pooling）减少特征图的尺寸，从而逐步捕获图像中的特征层次结构。

    第一层卷积

    1. nn.Conv2d(3, 16, kernel_size=3, stride=1, padding=1)：输入 3 通道（RGB），输出 16 个特征图，卷积核大小为 3x3，步长为 1，边界填充为 1。
    2. ReLU 激活函数，非线性映射将所有负值变为 0，正值保持不变，增加模型的非线性表达能力。
    3. nn.MaxPool2d(kernel_size=2, stride=2)：池化层，每 2x2 的区域取最大值，输出尺寸缩小为输入尺寸的 ½。

    第二层卷积

    将 16 个输入特征图卷积为 32 个输出特征图，依旧保持卷积核大小、步长和填充方式，并通过池化层缩小尺寸。

    第三层卷积

    将 32 个特征图卷积为 64 个输出特征图，再次池化。最终输出的特征图大小为 8x8（原始图像为 64x64，通过 3 层池化后缩小 8 倍）。

    全连接

    卷积层提取的特征图最终展平为一维特征向量，并通过全连接层进行分类。

    • 第一层全连接层：输入 64 * 8 * 8 个特征（5120 维），输出 128 维特征。
    • 第二层全连接层：输入 128 维特征，输出与类别数量相同的神经元。这样，最后输出的向量表示每个类别的分数，通过进一步的 softmax 函数可得到各类别的概率。

    训练过程

    • Epoch：一个完整的 epoch 意味着模型处理了所有的训练数据一次。
    • Forward Pass：将一批图像数据 images 传入模型，计算输出 outputs。
    • Loss Calculation：使用损失函数计算预测结果与真实标签的差距。
    • Backpropagation：计算梯度并更新权重，使模型误差最小化。
    • Loss Tracking：打印每轮的平均损失，用于评估模型的训练效果。
